Tuesday November 29 1994 16:01, root@sc22.perm.su (2:50/128) wrote to All:

r>    Возникла тут такая проблемка: пишу утилитку, которая выдает на
r> экран несколько страниц хелпа с ожиданием нажатия клавиши после
r> каждой страницы. В ней модуль CRT _не_ используется, поэтому
r> можно переназначить средствами MS_DOS'а вывод в файл. Так вот,
r> очень хотелось бы, чтобы при выводе в файл программа не ожидала
r> нажатия клавиши, а кидала бы все подряд (на манер ARJ). Вот в
r> этом-то собственно и проблема: как определить, переназначен вывод
r> в файл или нет.



function IsCON:Boolean; assembler;
asm
                 mov   ax,4400h
                 mov   bx,word ptr Output
                 int   21h
                 jnc   @@NoError

                 mov   IOResult,ax
                 jmp   @@Exit

@@NoError:       test  dx,2
                 jz    @@Exit

                 mov   al,1
@@Exit:
end;

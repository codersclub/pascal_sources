unit video;
interface
procedure screenoff;
procedure screenon;
implementation
procedure screenoff;assembler;
asm
     mov    dx,03cch
     in     al,dx
     and    al,11111101b
     mov    dx,03c2h
     out    dx,al
end;
procedure screenon;assembler;
asm
     mov    dx,03cch
     in     al,dx
     or     al,00000010b
     mov    dx,03c2h
     out    dx,al
end;
end.

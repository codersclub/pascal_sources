uses dos, crt;
const
  mask: array[1..4] of byte = ($10, $20, $40, $80);
  stat: array[boolean] of string[3] = ('On ', 'Off');
var
  r: registers;
  i: integer;
begin
  clrscr;
  writeln('Joystick tester, v0.1 by LSA 1997');
  repeat
    gotoxy(1, 3);
    r.ah := $84;
    r.dx := 0;
    intr($15, r);
    for i := 1 to 4 do
      writeln('Button ', i:1, ' = ', stat[r.al and mask[i] = mask[i]]);
    r.ah := $84;
    r.dx := 1;
    intr($15, r);
    writeln('     A.X = ', r.ax:3);
    writeln('     A.Y = ', r.bx:3);
    writeln('     B.X = ', r.cx:3);
    writeln('     B.Y = ', r.dx:3);
  until keypressed;
  readkey;
end.

uses Dos, Crt;

var
  code: byte;
  password: array[0..5] of byte;

function DeCode (symbol: byte): byte; assembler;
asm
  xor  bl,bl
  mov  al,symbol
@@cont:
  test    al,$80
  jz  @@c1
  test al,$61
  jnp  @@c2
  stc
@@c2:
  rcl      al,1
  jmp  @@check
@@c1:
  test al,$61
  jp  @@c3
  stc
@@c3:
  rcl  al,1
@@check:
  inc  bl
  cmp  al,code
  jne  @@cont
  xor  ax,ax
  mov  al,bl
end;

function ReadByte(address: byte): byte; assembler;
asm
  xor ax,ax
  mov al,address
  out 70H,al
  in al,71H
end;

procedure GetPassword;
var
  i: byte;
  b: byte;
begin
  code:=(ReadByte($37) and $F0);
  for i:=0 to 5 do
    begin
      b:=ReadByte($38+i);
      if b=0 then exit;
      password[i]:=DeCode(b);
      code:=b
    end
end;

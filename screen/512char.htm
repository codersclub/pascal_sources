 Вот:

       Mov ax,3
       Int 10h

       Mov ah,11h      ; знакогенератор
       Mov al,0        ; загр. табл
       Mov bx,1000h    ;N блока во 2 битовой плоскости
       Mov cx,0ffh     ;кол-во симв в табл.
       Mov dl,0;       ; код первого знака    -+
       Mov dh,0        ;-----------------------+

       ; ES:BP адр начала табл.
       Push cs
       Pop es
       Mov bp,offset Tabl0
       Int 10h

       Mov ah,11h      ; знакогенератор
       Mov al,0        ; загр. табл
       Mov bx,1001h    ;N блока во 2 битовой плоскости
       Mov cx,0ffh     ;кол-во симв в табл.
       Mov dl,0;       ; код первого знака    -+
       Mov dh,0        ;-----------------------+

       Push cs
       Pop es
       Mov bp,offset Tabl1
       Int 10h

 Это kусоk kода из моей шpифтозагpужалkи. Tabl0 и Tabl1 - адpеса шpифтовых
таблиц 8x16, kомментаpии есть, и пpедельно все пpосто.
